STREAMERBOT INTEGRATION
================================================================================

In Streamerbot, verwende "Execute Code" Action mit C#:

    using System;
    using System.IO;

    public bool Execute()
    {
        var commandsDir = @"C:\Streaming\Code\Minecraft\server\chaos-commands\";
        var timestamp = DateTime.Now.ToString("yyyyMMdd_HHmmss_fff");
        var filename = commandsDir + "cmd_" + timestamp + ".json";

        // Beispiel: 10 Chaos bei Follow
        var json = "{\"command\":\"add_chaos\",\"amount\":10,\"source\":\"follow\"}";

        File.WriteAllText(filename, json);
        return true;
    }

WICHTIG: Passe den Pfad in commandsDir an deinen Server-Pfad an!
